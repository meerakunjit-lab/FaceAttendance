<script>
function loadEmployees(){
  fetch('/api/employees')
    .then(res => res.json())
    .then(data => {
      const body = document.getElementById('emp-body');
      body.innerHTML = '';

      if(data.length === 0){
        document.getElementById('status').innerText = "No employees found";
        return;
      }

      document.getElementById('status').innerText =
        `Total Employees: ${data.length}`;

      data.forEach(emp => {
        const tr = document.createElement('tr');
        tr.innerHTML = `
          <td>${emp.emp_id}</td>
          <td>${emp.name}</td>
          <td>${emp.branch}</td>
          <td>${emp.phone || '-'}</td>
          <td>
            <button class="btn edit" onclick="editEmp(${emp.id})">Edit</button>
            <button class="btn delete" onclick="deleteEmp(${emp.id})">Delete</button>
          </td>
        `;
        body.appendChild(tr);
      });
    })
    .catch(() => {
      document.getElementById('status').innerText = "Failed to load employees";
    });
}

function editEmp(id){
  window.location.href = `/edit_employee/${id}`;
}

function deleteEmp(id){
  if(!confirm("Are you sure to delete this employee?")) return;

  fetch(`/delete_employee/${id}`, { method:'POST' })
    .then(res=>res.json())
    .then(data=>{
      alert(data.message);
      loadEmployees();
    });
}

loadEmployees();
